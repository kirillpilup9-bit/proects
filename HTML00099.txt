let currentPage = 0;

function renderBook() {
    const book = document.getElementById('book');
    book.innerHTML = '';

    const leftPage = document.createElement('div');
    leftPage.className = 'page left';
    leftPage.innerHTML = pages[currentPage] || '<p>Nav vairāk lapu.</p>';

    const rightPage = document.createElement('div');
    rightPage.className = 'page right';
    rightPage.innerHTML = pages[currentPage + 1] || '<p>Grāmatas beigas.</p>';

    book.appendChild(leftPage);
    book.appendChild(rightPage);
}

function nextPage() {
    if(currentPage + 2 < pages.length) {
        currentPage += 2;
        renderBook();
    }
}

function prevPage() {
    if(currentPage - 2 >= 0) {
        currentPage -= 2;
        renderBook();
    }
}
